from time import sleep

from core.util import clear from core.scene import Scene from core.frame import Frame

from client.client import Client from shapes.rect import Rect

client = Client()

scene = Scene('Fantastic scene')

rect = Rect(client, 10, 10, 100, 100, 'orange', 'transparent') scene.addshape(rect, 'orange-rect')

rect.v = [0, 15] rect.a = [0, 0]

class MoveFrame(Frame): def _init_(self): super()._init_()

def applyframe(self): rect = self.scene.shapes['orange-rect']

if rect.x <= 350: rect.x += rect.v[0]

if rect.y <= 350: rect.y += rect.v[1]

rect.v[0] += rect.a[0] rect.v[1] += rect.a[1]

for _ in range(24): frame = MoveFrame() scene.addframe(frame)

for _ in scene.frames: clear(client, True) scene.render() scene.nextframe(.05)

sleep(5)

clear(client)

client.close()
